
```
//计数器
var index = 0;
//定义一个函数，将类似的逻辑封装起来，并递归调用
var f = function ()
{
    //获取当前需要处理的元素
    var file = files[index++];
    file.div.html("正在保存文件");
    //异步方法
    window.upload([file.result], {
        callback: function ()
        {
            file.div.html("完成");
            //当异步方法执行完后，若数组尚未遍历完，则递归；否则清空计数器
            if (index < files.length)
            {
                f();
            }
            else
            {
                files.length = 0;
            }
        },
        error: function (error)
        {
            file.div.html(error.message);
            //当异步方法执行完后，若数组尚未遍历完，则递归；否则清空计数器
            if (index < files.length)
            {
                f();
            }
            else
            {
                files.length = 0;
            }
        }
    });
};
//执行递归函数
f();
```
